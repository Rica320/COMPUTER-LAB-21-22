101void draw_text(char *text, int x0, int y0, int hexColor, bool small) {
102 if (f_small == NULL) {
103 a = make_sprite(font_small, XPM_8_8_8_8);
104 f_small = create_animSprite(a, 96, 16, 11, 27);
105 }
106
107 if (small) {
108 int margin = 0;
109 for (unsigned i = 0; i < strlen(text); i++) {
110 int l = (int) (text[i]) - 32;
111
112 if (l < 0) {
113 return;
114 }
115
116 draw_animSprite(f_small, l, x0 + margin, y0);
117 margin += 10;
118 }
119 return;
120 }
121
122 for (unsigned i = 0; i < strlen(text); i++) {
123
124 int l = (int) (text[i]);
125
126 if (l == 32) {
127 x0 += 40;
128 continue;
129 }
130 else if (l == 58)
131 draw_char(xpm_char_058, x0, y0, hexColor);
132 else if (l == 47) // /
133 draw_char(xpm_char_047, x0, y0, hexColor);
134 else if (l >= 65 && l <= 90)
135 draw_char(letters[l - 65], x0, y0, hexColor);
136 else if (l >= 48 && l <= 57)
137 draw_char(numbers[l - 48], x0, y0, hexColor);
138 else
139 draw_char(xpm_char_035, x0, y0, hexColor);
140 x0 += 40;
141 }
142}
