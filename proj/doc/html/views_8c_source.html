537void updateTimer(bool white) {
538
539 // if game ended, stop moving clock
540 if (gameStateFlag)
541 return;
542
543 // if second has passed
544 if (rtc_data[0] - startTime) {
545
546 // set the new second time
547 startTime = rtc_data[0];
548
549 // decrement the current player's turn clock
550 if (white)
551 white_clock--;
552 else
553 black_clock--;
554 }
555
556 // if timer reached zero, raises game flag
557 if (white_clock == 0)
558 gameStateFlag = 2;
559
560 if (black_clock == 0)
561 gameStateFlag = 1;
562}
